64 std::shared_ptr<void> GetSignalkPayload(ObservedEvt ev) {
65  auto msg = UnpackEvtPointer<SignalkMsg>(ev);
66  wxJSONReader reader;
67  wxJSONValue data;
68  reader.Parse(wxString(msg->raw_message), &data);
69 
70  wxJSONValue root(wxJSONTYPE_OBJECT);
71  root["Data"] = data;
72  root["ErrorCount"] = reader.GetErrorCount();
73  root["WarningCount"] = reader.GetWarningCount();
74 
75  root["Errors"] = wxJSONValue(wxJSONTYPE_ARRAY);
76  for (size_t i = 0; i < reader.GetErrors().GetCount(); i++)
77  root["Errors"].Append(reader.GetErrors().Item(i));
78 
79  root["Warnings"] = wxJSONValue(wxJSONTYPE_ARRAY);
80  for (size_t i = 0; i < reader.GetWarnings().GetCount(); i++)
81  root["Warnings"].Append(reader.GetWarnings().Item(i));
82 
83  root["Context"] = msg->context;
84  root["ContextSelf"] = msg->context_self;
85 
86  return static_pointer_cast<void>(std::make_shared<wxJSONValue>(root));
87 }

158 CommDriverResult WriteCommDriver(
159  DriverHandle handle, const std::shared_ptr<std::vector<uint8_t>>& payload) {
160  // Find the driver from the handle
161  auto& registry = CommDriverRegistry::GetInstance();
162  auto drivers = registry.GetDrivers();
163  auto func = [handle](const DriverPtr d) { return d->Key() == handle; };
164  auto found = std::find_if(drivers.begin(), drivers.end(), func);
165 
166  if (found == drivers.end()) {
167  return RESULT_COMM_INVALID_HANDLE;
168  }
169  auto driver = *found;
170 
171  // Determine protocol
172  std::unordered_map<std::string, std::string> attributes =
173  GetAttributes(handle);
174  auto protocol_it = attributes.find("protocol");
175  if (protocol_it == attributes.end()) return RESULT_COMM_INVALID_PARMS;
176  std::string protocol = protocol_it->second;
177 
178  if (protocol == "nmea0183") {
179  auto d0183 = std::dynamic_pointer_cast<CommDriverN0183>(driver);
180 
181  std::string msg(payload->begin(), payload->end());
182  std::string id = msg.substr(1, 5);
183  auto address = std::make_shared<NavAddr0183>(d0183->iface);
184  auto msg_out = std::make_shared<Nmea0183Msg>( id, msg, address);
185  bool xmit_ok = d0183->SendMessage(msg_out, address);
186  return xmit_ok ? RESULT_COMM_NO_ERROR : RESULT_COMM_TX_ERROR;
187  } else
188  return RESULT_COMM_INVALID_PARMS;
189 }

191 CommDriverResult WriteCommDriverN2K(
192  DriverHandle handle, int PGN, int destinationCANAddress, int priority,
193  const std::shared_ptr<std::vector<uint8_t>>& payload) {
194  uint64_t _PGN;
195  _PGN = PGN;
196 
197  // Find the driver from the handle
198  auto& registry = CommDriverRegistry::GetInstance();
199  auto drivers = registry.GetDrivers();
200  auto func = [handle](const DriverPtr d) { return d->Key() == handle; };
201  auto found = std::find_if(drivers.begin(), drivers.end(), func);
202 
203  if (found == drivers.end()) {
204  return RESULT_COMM_INVALID_HANDLE;
205  }
206  auto driver = *found;
207  auto dest_addr = std::make_shared<const NavAddr2000>(driver->iface,
208  destinationCANAddress);
209  auto msg = std::make_shared<const Nmea2000Msg>(_PGN, *payload, dest_addr,
210  priority);
211  bool result = driver->SendMessage(msg, dest_addr);
212 
213  return RESULT_COMM_NO_ERROR;
214 }

216 CommDriverResult RegisterTXPGNs(DriverHandle handle,
217  std::vector<int>& pgn_list) {
218  if (!pgn_list.size()) return RESULT_COMM_INVALID_PARMS;
219 
220  // Find the driver from the handle
221  auto& registry = CommDriverRegistry::GetInstance();
222  auto drivers = registry.GetDrivers();
223  auto func = [handle](const DriverPtr d) { return d->Key() == handle; };
224  auto driver = std::find_if(drivers.begin(), drivers.end(), func);
225 
226  if (driver == drivers.end()) {
227  return RESULT_COMM_INVALID_HANDLE;
228  }
229  auto dn2k = std::dynamic_pointer_cast<CommDriverN2K>(*driver);
230 
231  int nloop = 0;
232  for (size_t i = 0; i < pgn_list.size(); i++) {
233  int nTry = 5;
234  int iresult = -1;
235  nloop = 0;
236  while (nTry && iresult < 0) {
237  iresult = dn2k->SetTXPGN(pgn_list[i]);
238  nTry--;
239  nloop++;
240  }
241  if (iresult < 0) {
242  return RESULT_COMM_REGISTER_PGN_ERROR;
243  }
244  }
245  return RESULT_COMM_NO_ERROR;
246 }
